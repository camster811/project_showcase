Finance Management Application

A personal finance tracking solution showcasing versatility in software development through two distinct implementations: a JavaFX desktop application and a Go-based web application. Both versions provide complete transaction management, financial summaries, and persistent data storage.

Java Transaction Model

public class Transaction implements Serializable {
    private int id;
    private String type;
    private double amount;
    private String description;

    public Transaction(int id, String type, double amount, String description) {
        this.id = id;
        this.type = type;
        this.amount = amount;
        this.description = description;
    }

    // Getters and setters...
    
    @Override
    public String toString() {
        return "Transaction " + id + "\nType: " + type + 
               "\nAmount: " + amount + "\nDescription: " + description;
    }
}

Go HTTP Handler

func (h *Handler) AddTransactionHandler(w http.ResponseWriter, r *http.Request) {
    if r.Method == "POST" {
        // Parse form data
        transactionType := r.FormValue("type")
        amountStr := r.FormValue("amount")
        description := r.FormValue("description")
        
        // Convert amount to float
        amount, err := strconv.ParseFloat(amountStr, 64)
        if err != nil {
            http.Error(w, "Invalid amount", http.StatusBadRequest)
            return
        }
        
        // Create and add transaction
        transaction := models.NewTransaction(transactionType, amount, description)
        h.financeManager.AddTransaction(transaction)
        
        // Redirect to home page
        http.Redirect(w, r, "/", http.StatusSeeOther)
    }
}

Go JSON Storage

func (fm *FinanceManager) SaveTransactions() error {
    data, err := json.MarshalIndent(fm.Transactions, "", "  ")
    if err != nil {
        return err
    }
    
    return ioutil.WriteFile(fm.FilePath, data, 0644)
}

func (fm *FinanceManager) LoadTransactions() error {
    data, err := ioutil.ReadFile(fm.FilePath)
    if err != nil {
        if os.IsNotExist(err) {
            // File doesn't exist, start with empty slice
            fm.Transactions = []Transaction{}
            return nil
        }
        return err
    }
    
    return json.Unmarshal(data, &fm.Transactions)
}
